package es.ies.puerto.modelo.db.dao.abstrac;

import java.sql.ResultSet;
import java.sql.Statement;

import es.ies.puerto.exception.MarvelException;
import es.ies.puerto.modelo.db.Conexion;

public class DaoUtilities extends Conexion{

    public DaoUtilities() throws MarvelException{
    }

    public void ejecutar(String sql)throws MarvelException{
        Statement statement = null;
    try {
        statement = getConexion().createStatement();
        statement.executeUpdate(sql);
    } catch (Exception e) {
        throw new MarvelException(e.getMessage(), e);  
    }finally{
        cerrar(statement, null);
        }
    }

    public void cerrar(Statement statement, ResultSet resultSet) throws MarvelException{
        try {
            if (resultSet!=null && !resultSet.isClosed()) {
                resultSet.close();
            }
            if (statement!=null && !statement.isClosed()) {
                statement.close();
            }
            if (getConexion()!=null && !getConexion().isClosed()) {
                getConexion().close();
            }
        } catch (Exception e) {
            throw new MarvelException(e.getMessage(),e);
        }
    }
}
