package es.ies.puerto.modelo.db.dao;

import es.ies.puerto.exception.MarvelException;
import es.ies.puerto.modelo.db.dao.abstrac.DaoUtilities;
import es.ies.puerto.modelo.db.entidades.Equipamiento;
import es.ies.puerto.modelo.db.entidades.Personaje;

import java.sql.ResultSet;
import java.sql.Statement;
import java.util.HashSet;
import java.util.Set;

public class DaoEquipamiento extends DaoUtilities {


    private static final String ID = "id";
    private static final String PERSONAJE_ID = "personaje_id";
    private static final String DESCRIPCION = "descripcion";
    private static final String NOMBRE = "nombre";

    public DaoEquipamiento() throws MarvelException {

    }

    public Set<Equipamiento> findAllEquipamiento() throws MarvelException {
        Statement statement = null;
            ResultSet resultSet = null;
            Set<Equipamiento> equipamientos = new HashSet<>();
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT id,nombre,descripcion,personaje_id FROM Equipamiento");
                while (resultSet.next()) {
                    String id=resultSet.getString(ID);
                    String nombre=resultSet.getString(NOMBRE);
                    String descripcion=resultSet.getString(DESCRIPCION);
                    String personajeId=resultSet.getString(PERSONAJE_ID);
                    equipamientos.add(new Equipamiento(id, nombre, descripcion, personajeId));
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return equipamientos;
    }

    public Equipamiento findEquipamiento(Equipamiento equipamiento) throws MarvelException {
        Statement statement = null;
            ResultSet resultSet = null;
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT id,nombre,descripcion,personaje_id FROM Equipamiento WHERE id='"+equipamiento.getId()+"'");
                while (resultSet.next()) {
                    String id=resultSet.getString(ID);
                    String nombre=resultSet.getString(NOMBRE);
                    String descripcion=resultSet.getString(DESCRIPCION);
                    String personajeId=resultSet.getString(PERSONAJE_ID);
                    return new Equipamiento(id, nombre, descripcion, personajeId);
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return null;
    }

    public boolean updateEquipamiento(Equipamiento equipamiento) throws MarvelException {
        if (equipamiento==null) {
            return false;
        }
        if (findAllEquipamiento().contains(equipamiento)) {
            try {
                ejecutar("UPDATE equipamiento SET personaje_id='"+equipamiento.getPersonaje_id()+"', nombre='"+equipamiento.getNombre()+"', descripcion='"+equipamiento.getDescripcion()+"' WHERE id='"+equipamiento.getId()+"'");
                return true;
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }
        }
        try {
            String agregarEquipamientoQuery= "INSERT INTO Equipamiento (id,nombre,descripcion,personaje_id) VALUES ('"+equipamiento.getId()+"','"+equipamiento.getNombre()+"','"+equipamiento.getDescripcion()+"','"+equipamiento.getPersonaje_id()+"')";
            ejecutar(agregarEquipamientoQuery);
            return true;
        } catch (Exception e) {
            throw new MarvelException(e.getMessage(),e);
        }
    }

    public void deleteEquipamiento(Equipamiento equipamiento) throws MarvelException {
        if (findAllEquipamiento().contains(equipamiento)) {
            ejecutar("DELETE FROM equipamiento WHERE id='"+equipamiento.getId()+"'");
        }
    }

    public Set<Equipamiento> findEquipamientosPersonaje(Personaje personaje)throws MarvelException{
        Statement statement = null;
            ResultSet resultSet = null;
            Set<Equipamiento> equipamientos = new HashSet<>();
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT id,nombre,descripcion,personaje_id FROM Equipamiento WHERE personaje_id='"+personaje.getId()+"'");
                while (resultSet.next()) {
                    String id=resultSet.getString(ID);
                    String nombre=resultSet.getString(NOMBRE);
                    String descripcion=resultSet.getString(DESCRIPCION);
                    String personajeId=resultSet.getString(PERSONAJE_ID);
                    equipamientos.add(new Equipamiento(id, nombre, descripcion, personajeId));
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return equipamientos;
    }


}
