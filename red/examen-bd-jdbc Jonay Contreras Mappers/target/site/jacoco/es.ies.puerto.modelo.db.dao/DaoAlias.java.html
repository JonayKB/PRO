package es.ies.puerto.modelo.db.dao;

import es.ies.puerto.exception.MarvelException;
import es.ies.puerto.modelo.db.dao.abstrac.DaoUtilities;
import es.ies.puerto.modelo.db.entidades.Alias;
import es.ies.puerto.modelo.db.entidades.Personaje;
import java.sql.ResultSet;
import java.sql.Statement;
import java.util.HashSet;
import java.util.Set;

public class DaoAlias  extends DaoUtilities{


    private static final String ALIAS = "alias";
    private static final String ID = "id";
    private static final String PERSONAJE_ID = "personaje_id";

    public DaoAlias() throws MarvelException {

        }

        

        public Set<Alias> findAllAlias() throws MarvelException {
            Statement statement = null;
            ResultSet resultSet = null;
            Set<Alias> alias = new HashSet<>();
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT id,personaje_id,alias FROM Alias");
                while (resultSet.next()) {
                    String id=resultSet.getString(ID);
                    String personajeId=resultSet.getString(PERSONAJE_ID);
                    String descipcion=resultSet.getString(ALIAS);
                    alias.add(new Alias(id, descipcion, personajeId));
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return alias;
        }

        public Alias findAlias(Alias alias) throws MarvelException {
            Statement statement = null;
            ResultSet resultSet = null;
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT id,personaje_id,alias FROM alias WHERE id='"+alias.getId()+"'");
                while (resultSet.next()) {
                    String id=resultSet.getString(ID);
                    String personajeId=resultSet.getString(PERSONAJE_ID);
                    String descripcion=resultSet.getString(ALIAS);
                    return new Alias(id, descripcion, personajeId);
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return null;
        }

        public boolean updateAlias(Alias alias) throws MarvelException {
            if (alias==null) {
                return false;
            }
            if (findAllAlias().contains(alias)) {
                try {
                    ejecutar("UPDATE alias SET personaje_id='"+alias.getPersonaje_id()+"', alias='"+alias.getDescripcion()+"' WHERE id='"+alias.getId()+"'");
                    return true;
                } catch (Exception e) {
                    throw new MarvelException(e.getMessage(),e);
                }
            }
            try {
                String agregarAliasQuery= "INSERT INTO Alias (id,alias,personaje_id) VALUES ('"+alias.getId()+"','"+alias.getDescripcion()+"','"+alias.getPersonaje_id()+"')";
                ejecutar(agregarAliasQuery);
                return true;
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }
            
        }

        public void deleteAlias(Alias alias) throws MarvelException {
            if (findAllAlias().contains(alias)) {
                ejecutar("DELETE FROM alias WHERE id='"+alias.getId()+"'");
            }
        }

        public Alias findAliasPersonaje(Personaje personaje)throws MarvelException{
            Statement statement = null;
            ResultSet resultSet = null;
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT id,personaje_id,alias FROM alias WHERE personaje_id='"+personaje.getId()+"'");
                while (resultSet.next()) {
                    String id=resultSet.getString(ID);
                    String personajeId=resultSet.getString(PERSONAJE_ID);
                    String alias=resultSet.getString(ALIAS);
                    return new Alias(id, alias, personajeId);
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return null;
        }


}
