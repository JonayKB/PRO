package es.ies.puerto.modelo.db.dao;

import es.ies.puerto.exception.MarvelException;
import es.ies.puerto.modelo.db.dao.abstrac.DaoUtilities;
import es.ies.puerto.modelo.db.entidades.Equipamiento;
import es.ies.puerto.modelo.db.entidades.Personaje;
import es.ies.puerto.modelo.db.entidades.Poder;
import java.sql.ResultSet;
import java.sql.Statement;
import java.util.HashSet;
import java.util.Set;

public class DaoPersonaje  extends DaoUtilities{
    private static final String PERSONAJE_ID = "personaje_id";
    private static final String PODER_ID = "poder_id";
    private static final String GENERO = "genero";
    private static final String NOMBRE = "nombre";
    private static final String ID = "id";
    DaoAlias daoAlias;
    DaoEquipamiento daoEquipamiento;
    DaoPoder daoPoder;
    public DaoPersonaje() throws MarvelException {
        daoAlias = new DaoAlias();
        daoEquipamiento = new DaoEquipamiento();
        daoPoder = new DaoPoder();
    }

    public Set<Personaje> findAllPersonaje() throws MarvelException {
        Statement statement = null;
            ResultSet resultSet = null;
            Set<Personaje> personajes = new HashSet<>();
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT id,nombre,genero FROM personaje");
                while (resultSet.next()) {
                    String id=resultSet.getString(ID);
                    String nombre=resultSet.getString(NOMBRE);
                    String genero=resultSet.getString(GENERO);
                    Personaje personaje = new Personaje(id, nombre, genero, null, null, null);
                    personaje.setAlias(daoAlias.findAliasPersonaje(personaje));
                    personaje.setEquipamientos(daoEquipamiento.findEquipamientosPersonaje(personaje));
                    personaje.setPoderes(daoPoder.findPoderesPersonaje(personaje));
                    personajes.add(personaje);
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return personajes;
    }

    public Personaje findPersonaje(Personaje personaje) throws MarvelException {
        Statement statement = null;
            ResultSet resultSet = null;
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT id,nombre,genero FROM personaje WHERE id='"+personaje.getId()+"'");
                while (resultSet.next()) {
                    String id=resultSet.getString(ID);
                    String nombre=resultSet.getString(NOMBRE);
                    String genero=resultSet.getString(GENERO);
                    personaje = new Personaje(id, nombre, genero, null, null, null);
                    personaje.setAlias(daoAlias.findAliasPersonaje(personaje));
                    personaje.setEquipamientos(daoEquipamiento.findEquipamientosPersonaje(personaje));
                    personaje.setPoderes(daoPoder.findPoderesPersonaje(personaje));
                    return personaje;
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return null;
    }

    public boolean updatePersonaje(Personaje personaje) throws MarvelException {
        if (personaje==null) {
            return false;
        }
        String qrry ="INSERT INTO Personaje (id,nombre,genero) VALUES ('"+personaje.getId()+"','"+personaje.getNombre()+"','"+personaje.getGenero()+"')";
        if (findAllPersonaje().contains(personaje)) {
            qrry="UPDATE personaje SET nombre='"+personaje.getNombre()+"', genero='"+personaje.getGenero()+"' WHERE id='"+personaje.getId()+"'";
        }
            try {
                ejecutar(qrry);
                daoAlias.updateAlias(personaje.getAlias());
                if (personaje.getPoderes()!=null) {
                    for (Poder poder : personaje.getPoderes()) {
                        daoPoder.updatePoder(poder);
                        updatePersonajePoder(personaje, poder);
                    }
                }
                if (personaje.getEquipamientos()!=null) {
                    for (Equipamiento equipamiento : personaje.getEquipamientos()) {
                        daoEquipamiento.updateEquipamiento(equipamiento);
                    }
                }
                
                
                return true;
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }
    }

    public void deletePersonaje(Personaje personaje) throws MarvelException {
        if (findAllPersonaje().contains(personaje)) {
            ejecutar("DELETE FROM personaje WHERE id='"+personaje.getId()+"'");
            if (personaje.getEquipamientos()!=null) {
                for (Equipamiento equipamiento : personaje.getEquipamientos()) {
                    daoEquipamiento.deleteEquipamiento(equipamiento);
                }
            }
            if (personaje.getPoderes()!=null) {
                for (Poder poder : personaje.getPoderes()) {
                    deletePersonajePoder(personaje, poder);
                }
            }
            daoAlias.deleteAlias(personaje.getAlias());
            deletePoderNoPersonaje();
            
        }
    }


    public Set<String> findAllPersonajePoder() throws MarvelException{
        Statement statement = null;
            ResultSet resultSet = null;
            Set<String> personajePoder = new HashSet<>();
            try {
                statement=getConexion().createStatement();
                resultSet=statement.executeQuery("SELECT personaje_id,poder_id FROM Personaje_poder");
                while (resultSet.next()) {
                    String poderId=resultSet.getString(PODER_ID);
                    String personajeId=resultSet.getString(PERSONAJE_ID);
                    personajePoder.add(personajeId+" "+poderId);
                }
            } catch (Exception e) {
                throw new MarvelException(e.getMessage(),e);
            }finally{
                cerrar(statement, resultSet);
            }
            return personajePoder;
    }

    public boolean updatePersonajePoder(Personaje personaje, Poder poder)throws MarvelException{
        if (findAllPersonajePoder().contains(personaje.getId()+" "+poder.getId())) {
            return true;
        }
        try {
            String agregarPersonajePoder= "INSERT INTO personaje_poder (personaje_id,poder_id) VALUES ('"+personaje.getId()+"','"+poder.getId()+"')";
            ejecutar(agregarPersonajePoder);
            return true;
        } catch (Exception e) {
            throw new MarvelException(e.getMessage(),e);
        }
    }

    public void deletePersonajePoder(Personaje personaje, Poder poder) throws MarvelException{
        if (findAllPersonajePoder().contains(personaje.getId()+" "+poder.getId())) {
            ejecutar("DELETE FROM personaje_poder WHERE personaje_id='"+personaje.getId()+"' AND poder_id='"+poder.getId()+"'");
        }
    }

    

    public void deletePoderNoPersonaje()throws MarvelException{
        String selecion = "SELECT p.id FROM poder as p LEFT JOIN personaje_poder pp ON p.id=pp.poder_id WHERE pp.personaje_id IS NULL";
        ejecutar("DELETE FROM poder WHERE id IN("+selecion+")");
    }
